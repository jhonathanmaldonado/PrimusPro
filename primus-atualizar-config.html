2. Regras do Firestore Manual

O erro "Missing or insufficient permissions" nas telas de Escala e Fechamento é causado pelas regras do Firestore. Copie e cole as regras abaixo no console do Firebase.

Como fazer:
1. Acesse Firebase Console → Firestore → Rules
2. Apague tudo e cole o conteúdo abaixo
3. Clique em Publicar 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Função auxiliar: obtém o perfil do usuário logado
    function perfil() {
      return get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.perfil;
    }
    function ativo() {
      return get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.ativo == true;
    }
    function autenticado() {
      return request.auth != null && ativo();
    }
    function isGerente()   { return autenticado() && perfil() in ['gerente','admin']; }
    function isFinanceiro(){ return autenticado() && perfil() in ['financeiro','diretoria','admin']; }
    function isDiretoria() { return autenticado() && perfil() in ['diretoria','admin']; }
    function isAdmin()     { return autenticado() && perfil() == 'admin'; }
    function qualquerPerfil() { return autenticado(); }

    // Usuários — admin lê e escreve todos; cada user lê o próprio
    match /usuarios/{uid} {
      allow read: if autenticado();
      allow write: if isAdmin();
      allow update: if request.auth.uid == uid; // usuário pode atualizar o próprio doc
    }

    // Funcionários — diretoria/admin escrevem; todos lêem
    match /funcionarios/{id} {
      allow read:  if qualquerPerfil();
      allow write: if isDiretoria() || isGerente(); // gerente pode criar via escala
    }

    // Presenças — gerente cria e edita (se em_aberto); financeiro lê
    match /presencas_diarias/{data} {
      allow read:  if qualquerPerfil();
      allow create: if isGerente();
      allow update: if isGerente() || isFinanceiro() || isDiretoria();
    }

    // Fechamento por dia — financeiro/diretoria/admin escrevem
    match /fechamento_diario/{id} {
      allow read:  if qualquerPerfil();
      allow write: if isFinanceiro();
    }

    // Fechamento por funcionário — financeiro/diretoria/admin escrevem
    match /fechamento_funcionario/{id} {
      allow read:  if qualquerPerfil();
      allow write: if isFinanceiro();
    }

    // Configurações (gorjeta) — admin escreve; todos lêem
    match /configuracoes/{id} {
      allow read:  if qualquerPerfil();
      allow write: if isAdmin();
    }

    // Escala semanal — gerente/admin escrevem; financeiro lê
    match /escala_semanal/{id} {
      allow read:  if qualquerPerfil();
      allow write: if isGerente();
    }

    // Auditoria — qualquer perfil logado pode criar; ninguém edita
    match /auditoria/{id} {
      allow read:  if isDiretoria();
      allow create: if qualquerPerfil();
      allow update, delete: if false;
    }
  }
}
